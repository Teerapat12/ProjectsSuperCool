# This document contains basic commands to use git on the command-line, shell, terminal.

#downloading a repository from git 
git clone "project_to_download"

#create a new branch 
git checkout -b "new_branch_name"

#how many branches are existing and in which branch am I atm 
git branch 

#pushing the new created branch into the repository

git push -u origin "the_name_of_the_new_created_branch"

#set you git commit credentials 

git config --global user.email "you@example.com"
git config --global user.name "Your Name"
